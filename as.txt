byte gmul(byte a,byte b)
{
byte p=0;
for(int i=0;i<8;i++)
{
if(b&1) p^=a;
bool hi=a&0x80;
a<<=1;
if(hi) a^=0x1b;
b>>=1;
}
return p;
}

/* SubBytes */
void SubBytes(byte s[4][4])
{
for(int i=0;i<4;i++)
for(int j=0;j<4;j++)
s[i][j]=sbox[s[i][j]];
}

/* ShiftRows */
void ShiftRows(byte s[4][4])
{
byte temp;

temp=s[1][0];
s[1][0]=s[1][1];
s[1][1]=s[1][2];
s[1][2]=s[1][3];
s[1][3]=temp;

swap(s[2][0],s[2][2]);
swap(s[2][1],s[2][3]);

temp=s[3][3];
s[3][3]=s[3][2];
s[3][2]=s[3][1];
s[3][1]=s[3][0];
s[3][0]=temp;
}

/* MixColumns */
void MixColumns(byte s[4][4])
{
byte t[4];

for(int i=0;i<4;i++)
{
t[0]=gmul(2,s[0][i])^gmul(3,s[1][i])^s[2][i]^s[3][i];
t[1]=s[0][i]^gmul(2,s[1][i])^gmul(3,s[2][i])^s[3][i];
t[2]=s[0][i]^s[1][i]^gmul(2,s[2][i])^gmul(3,s[3][i]);
t[3]=gmul(3,s[0][i])^s[1][i]^s[2][i]^gmul(2,s[3][i]);

for(int j=0;j<4;j++)
s[j][i]=t[j];
}
}

/* AddRoundKey */
void AddRoundKey(byte s[4][4],byte key[16])
{
for(int i=0;i<4;i++)
for(int j=0;j<4;j++)
s[j][i]^=key[i*4+j];
}

/* AES Encryption */
string encryptAES(string msg,string key)
{
byte state[4][4];

while(msg.size()<16) msg+=' ';

for(int i=0;i<16;i++)
state[i%4][i/4]=msg[i];

byte k[16];
for(int i=0;i<16;i++) k[i]=key[i];

AddRoundKey(state,k);

for(int round=1;round<=9;round++)
{
SubBytes(state);
ShiftRows(state);
MixColumns(state);
AddRoundKey(state,k);
}

SubBytes(state);
ShiftRows(state);
AddRoundKey(state,k);

string cipher="";

for(int i=0;i<4;i++)
for(int j=0;j<4;j++)
cipher+=state[j][i];

return cipher;
}

/* Convert to HEX */
string toHex(string s)
{
stringstream ss;
for(unsigned char c:s)
ss<<hex<<setw(2)<<setfill('0')<<(int)c;
return ss.str();
}

#include "aes.h"
#include <arpa/inet.h>
#include <unistd.h>
#include <iostream>

using namespace std;

int main()
{
    int clientsocket = socket(AF_INET, SOCK_STREAM, 0);

    sockaddr_in addr{};
    addr.sin_family = AF_INET;
    addr.sin_port = htons(8080);

    inet_pton(AF_INET, "127.0.0.1", &addr.sin_addr);

    connect(clientsocket, (sockaddr*)&addr, sizeof(addr));

    while(true)
    {
        string msg;

        cout << "Client message : ";
        getline(cin, msg);

        string cipher = encryptAES(msg);

        cout << "Cipher text : " << toHex(cipher) << endl;

        int len = cipher.length();

        send(clientsocket, &len, sizeof(len), 0);
        send(clientsocket, cipher.c_str(), len, 0);

        if(msg == "exit")
            break;

        int rlen;

        recv(clientsocket, &rlen, sizeof(rlen), 0);

        char buffer[1024];

        recv(clientsocket, buffer, rlen, 0);

        string reply(buffer, rlen);

        cout << "Encrypted reply : " << toHex(reply) << endl;

        string decrypted = decryptAES(reply);

        cout << "Decrypted reply : " << decrypted << endl;
    }

    close(clientsocket);
}
