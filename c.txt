sha.h

#ifndef SHA_H
#define SHA_H

#include <openssl/sha.h>
#include <string>
#include <sstream>
#include <iomanip>

using namespace std;

string sha256(string input)
{
    unsigned char hash[SHA256_DIGEST_LENGTH];

    SHA256((unsigned char*)input.c_str(), input.length(), hash);

    stringstream ss;

    for(int i=0;i<SHA256_DIGEST_LENGTH;i++)
        ss<<hex<<setw(2)<<setfill('0')<<(int)hash[i];

    return ss.str();
}

server.cpp (sha)

#include "sha.h"
#include <arpa/inet.h>
#include <unistd.h>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int serversocket = socket(AF_INET, SOCK_STREAM, 0);

    sockaddr_in addr{};
    addr.sin_family = AF_INET;
    addr.sin_port = htons(8080);
    addr.sin_addr.s_addr = INADDR_ANY;

    bind(serversocket,(sockaddr*)&addr,sizeof(addr));
    listen(serversocket,0);

    cout<<"Server listening...\n";

    int clientsocket = accept(serversocket,nullptr,nullptr);

    while(true)
    {
        int len;
        recv(clientsocket,&len,sizeof(len),0);

        if(len==0) break;

        char buffer[1024];
        recv(clientsocket,buffer,len,0);
        buffer[len]='\0';

        string msg = buffer;

        char hash_recv[200];
        recv(clientsocket,hash_recv,sizeof(hash_recv),0);

        string received_hash = hash_recv;

        cout<<"Message received : "<<msg<<endl;
        cout<<"Hash received : "<<received_hash<<endl;

        string server_hash = sha256(msg);

        cout<<"Hash computed : "<<server_hash<<endl;

        if(server_hash == received_hash)
            cout<<"Integrity Verified\n";
        else
            cout<<"Message Tampered\n";
    }

    close(clientsocket);
    close(serversocket);
}

client.cpp (sha)

#include "sha.h"
#include <arpa/inet.h>
#include <unistd.h>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int clientsocket = socket(AF_INET, SOCK_STREAM, 0);

    sockaddr_in addr{};
    addr.sin_family = AF_INET;
    addr.sin_port = htons(8080);

    inet_pton(AF_INET,"127.0.0.1",&addr.sin_addr);

    connect(clientsocket,(sockaddr*)&addr,sizeof(addr));

    while(true)
    {
        string msg;

        cout<<"Client message : ";
        getline(cin,msg);

        int len = msg.length();

        send(clientsocket,&len,sizeof(len),0);
        send(clientsocket,msg.c_str(),len,0);

        string hash = sha256(msg);

        cout<<"Generated SHA256 hash : "<<hash<<endl;

        send(clientsocket,hash.c_str(),hash.length()+1,0);

        if(msg=="exit")
            break;
    }

    close(clientsocket);
}
