string permute(const string &in, int *table, int n) {
string out(n,'0');
for(int i=0; i<n; i++) out[i]=in[table[i]-1];
return out;
}
string xorBits(const string &a,const string &b) {
string out=a;
for(int i=0; i<a.size(); i++)
out[i]=(a[i]==b[i])?'0':'1';
return out;
}
string leftShift(const string &s,int n) {

return s.substr(n)+s.substr(0,n);
}
string toBinary4(int v) {
string r="0000";
for(int i=3; i>=0; i--) {
r[i]=(v%2)+'0';
v/=2;
}
return r;
}
vector<string> generateKeys(string key64) {
string key56 = permute(key64,PC1,56);
string C = key56.substr(0,28);
string D = key56.substr(28,28);
vector<string> keys;
for(int i=0; i<16; i++) {
C = leftShift(C,SHIFTS[i]);
D = leftShift(D,SHIFTS[i]);
keys.push_back(permute(C+D,PC2,48));
}
return keys;
}
string feistel(const string &R,const string &K) {
string exp = permute(R,E,48);
string x = xorBits(exp,K);
string out="";
for(int i=0; i<8; i++) {
string b=x.substr(i*6,6);
int r=(b[0]-'0')*2+(b[5]-'0');
int c=(b[1]-'0')*8+(b[2]-'0')*4+(b[3]-'0')*2+(b[4]-'0');
out+=toBinary4(S[i][r][c]);
}
return permute(out,P,32);
}
string DES(string text, vector<string> keys) {
text = permute(text,IP,64);
string L=text.substr(0,32), R=text.substr(32,32);
for(int i=0; i<16; i++) {
string tmp=R;
R=xorBits(L,feistel(R,keys[i]));
L=tmp;

}
return permute(R+L,FP,64);
}
string desEncrypt(const string &plain,const string &key) {
return DES(plain,generateKeys(key));
}
string desDecrypt(const string &cipher,const string &key) {
auto k=generateKeys(key);
reverse(k.begin(),k.end());
return DES(cipher,k);
}

#include<iostream>
#include<bitset>
#include<cstring>
#include<unistd.h>
#include<netinet/in.h>
#include<sys/socket.h>
#include<sys/select.h>
#include "des.h"

using namespace std;


string textToBinary(string text) {
    int rem = text.length() % 8;
    if (rem) text.append(8 - rem, ' ');

    string binary="";
    for(char c:text)
        binary += bitset<8>(c).to_string();
    return binary;
}

string binaryToText(string binary) {
    string text="";
    for(int i=0;i<binary.length();i+=8) {
        bitset<8> b(binary.substr(i,8));
        text += char(b.to_ulong());
    }
    return text;
}


int main() {
    int serverSocket = socket(AF_INET, SOCK_STREAM, 0);

    sockaddr_in serverAddress{};
    serverAddress.sin_family = AF_INET;
    serverAddress.sin_port = htons(8080);
    serverAddress.sin_addr.s_addr = INADDR_ANY;

    bind(serverSocket,(sockaddr*)&serverAddress,sizeof(serverAddress));
    listen(serverSocket,5);

    cout << "=== SERVER STARTED ===\n";
    cout << "Type 'exit' anytime to stop server\n";

    string key = "1010110011011100100110011011010111001011100010101001000001010011";

    fd_set readfds;

    while (true)
    {
        FD_ZERO(&readfds);
        FD_SET(serverSocket, &readfds); // socket
        FD_SET(STDIN_FILENO, &readfds); // keyboard

        int maxfd = max(serverSocket, STDIN_FILENO) + 1;

        select(maxfd, &readfds, NULL, NULL, NULL);

       
        if (FD_ISSET(STDIN_FILENO, &readfds)) {
            string cmd;
            getline(cin, cmd);
            if (cmd == "exit") {
                cout << "Shutting down server...\n";
                close(serverSocket);
                return 0;
            }
        }

       
        if (FD_ISSET(serverSocket, &readfds)) {
            int clientSocket = accept(serverSocket, nullptr, nullptr);
            cout << "\nClient connected!\n";

            while (true)
            {
                char buffer[1024]={0};
                int len = recv(clientSocket, buffer, sizeof(buffer), 0);

                if(len <= 0) {
                    cout << "Client disconnected.\n";
                    break;
                }

                string encrypted(buffer, len);
                cout << "\n[CLIENT MESSAGE RECEIVED]\n";
                cout << "Received (Binary)  : " << encrypted << endl;
                cout << "Received (Text)    : " << binaryToText(encrypted) << endl;

                string decrypted="";
                for(int i=0;i<encrypted.length();i+=64)
                    decrypted += desDecrypt(encrypted.substr(i,64), key);

                cout << "Decrypted (Binary) : " << decrypted << endl;
                cout << "Decrypted (Text)   : " << binaryToText(decrypted) << endl;

                string reply;
                cout << "\nEnter reply: ";
                getline(cin, reply);

                string binary = textToBinary(reply);

                string encryptedReply="";
                for(int i=0;i<binary.length();i+=64)
                    encryptedReply += desEncrypt(binary.substr(i,64), key);

                send(clientSocket, encryptedReply.c_str(), encryptedReply.size(), 0);
            }

            close(clientSocket);
        }
    }
   
}

#include<iostream>

using namespace std;


string textToBinary(string text) {
    int rem = text.length() % 8;
    if (rem) text.append(8 - rem, ' ');

    string binary="";
    for(char c:text)
        binary += bitset<8>(c).to_string();
    return binary;
}

string binaryToText(string binary) {
    string text="";
    for(int i=0;i<binary.length();i+=8) {
        bitset<8> b(binary.substr(i,8));
        text += char(b.to_ulong());
    }
    return text;
}


int main() {
    int clientSocket = socket(AF_INET, SOCK_STREAM, 0);

    sockaddr_in serverAddress{};
    serverAddress.sin_family = AF_INET;
    serverAddress.sin_port = htons(8080);
    serverAddress.sin_addr.s_addr = inet_addr("127.0.0.1");

    connect(clientSocket, (sockaddr*)&serverAddress, sizeof(serverAddress));

    cout << "\n=== CLIENT STARTED ===\nConnected to server\n";

    string key = "1010110011011100100110011011010111001011100010101001000001010011";

    while (true) {
        string message;
        cout << "\nEnter message (English): ";
        getline(cin, message);

        if (message == "exit") break;

        cout << "\n[CLIENT SIDE]\n";
        cout << "Original Text      : " << message << endl;

        string binary = textToBinary(message);
        cout << "Binary             : " << binary << endl;

        string encrypted="";
        for(int i=0;i<binary.length();i+=64)
            encrypted += desEncrypt(binary.substr(i,64), key);

        cout << "Encrypted (Binary) : " << encrypted << endl;
        cout << "Encrypted (Text)   : " << binaryToText(encrypted) << endl;

        send(clientSocket, encrypted.c_str(), encrypted.size(), 0);

        // ---- RECEIVE REPLY ----
        char buffer[1024]={0};
        int len = recv(clientSocket, buffer, sizeof(buffer), 0);
        if(len <= 0) break;

        string encReply(buffer, len);

        cout << "\n[SERVER REPLY RECEIVED]\n";
        cout << "Encrypted (Binary) : " << encReply << endl;
        cout << "Encrypted (Text)   : " << binaryToText(encReply) << endl;

        string decrypted="";
        for(int i=0;i<encReply.length();i+=64)
            decrypted += desDecrypt(encReply.substr(i,64), key);

        cout << "Decrypted (Binary) : " << decrypted << endl;
        cout << "Decrypted (Text)   : " << binaryToText(decrypted) << endl;
    }

    close(clientSocket);
}
