rsa.h

#ifndef RSA_H
#define RSA_H

#include <bits/stdc++.h>
using namespace std;

/* Fast Modular Exponentiation */
long long modexp(long long base,long long exp,long long mod)
{
long long res=1;

while(exp>0)
{
if(exp&1)
res=(res*base)%mod;

base=(base*base)%mod;
exp>>=1;
}

return res;
}

/* GCD */
long long gcd(long long a,long long b)
{
if(b==0) return a;
return gcd(b,a%b);
}

/* Modular inverse */
long long modinv(long long e,long long phi)
{
for(long long d=1;d<phi;d++)
if((e*d)%phi==1)
return d;

return -1;
}

#endif

rsa_client.cpp

#include "rsa.h"
#include <arpa/inet.h>
#include <unistd.h>
#include <iostream>

using namespace std;

long long e,n;

int main()
{

int clientsocket=socket(AF_INET,SOCK_STREAM,0);

sockaddr_in addr{};
addr.sin_family=AF_INET;
addr.sin_port=htons(8080);

inet_pton(AF_INET,"127.0.0.1",&addr.sin_addr);

connect(clientsocket,(sockaddr*)&addr,sizeof(addr));

/* Receive public key */
recv(clientsocket,&e,sizeof(e),0);
recv(clientsocket,&n,sizeof(n),0);

while(true)
{

string msg;

cout<<"Client message : ";
getline(cin,msg);

int len=msg.length();

send(clientsocket,&len,sizeof(len),0);

for(char ch:msg)
{

long long cipher = modexp(ch,e,n);

cout<<"Encrypted sent : "<<cipher<<endl;

send(clientsocket,&cipher,sizeof(cipher),0);

}

if(msg=="exit") break;

int rlen;

recv(clientsocket,&rlen,sizeof(rlen),0);

string reply="";

for(int i=0;i<rlen;i++)
{

long long cipher;

recv(clientsocket,&cipher,sizeof(cipher),0);

cout<<"Encrypted received : "<<cipher<<endl;

char ch = modexp(cipher,e,n);

reply+=ch;

}

cout<<"Decrypted reply : "<<reply<<endl;

}

close(clientsocket);

}

rsa_server.cpp

#include "rsa.h"
#include <arpa/inet.h>
#include <unistd.h>
#include <iostream>

using namespace std;

long long p=61;
long long q=53;

long long n=p*q;
long long phi=(p-1)*(q-1);

long long e=17;
long long d;

int main()
{

d=modinv(e,phi);

int serversocket=socket(AF_INET,SOCK_STREAM,0);

sockaddr_in addr{};
addr.sin_family=AF_INET;
addr.sin_port=htons(8080);
addr.sin_addr.s_addr=INADDR_ANY;

bind(serversocket,(sockaddr*)&addr,sizeof(addr));
listen(serversocket,0);

cout<<"Server listening...\n";

int clientsocket=accept(serversocket,nullptr,nullptr);

/* Send public key */
send(clientsocket,&e,sizeof(e),0);
send(clientsocket,&n,sizeof(n),0);

while(true)
{

int len;

recv(clientsocket,&len,sizeof(len),0);

if(len==0) break;

string msg="";

for(int i=0;i<len;i++)
{

long long cipher;

recv(clientsocket,&cipher,sizeof(cipher),0);

cout<<"Encrypted received : "<<cipher<<endl;

char ch = modexp(cipher,d,n);

msg+=ch;
}

cout<<"Decrypted message : "<<msg<<endl;

if(msg=="exit") break;

string reply;

cout<<"Server reply : ";
getline(cin>>ws,reply);

int rlen=reply.length();

send(clientsocket,&rlen,sizeof(rlen),0);

for(char ch:reply)
{

long long cipher = modexp(ch,e,n);

cout<<"Encrypted sent : "<<cipher<<endl;

send(clientsocket,&cipher,sizeof(cipher),0);

}

}

close(clientsocket);
close(serversocket);

}
