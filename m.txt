#include <iostream>
#include <vector>
#include <cmath>
#include <arpa/inet.h>
#include <unistd.h>
#include <cstdio>

using namespace std;

typedef unsigned int uint32;

uint32 leftrotate(uint32 x, uint32 c) {
    return (x << c) | (x >> (32 - c));
}

string md5(string input) {
    vector<unsigned char> msg(input.begin(), input.end());

    uint64_t bit_len = msg.size() * 8;

    msg.push_back(0x80);

    while (msg.size() % 64 != 56)
        msg.push_back(0);

    for (int i = 0; i < 8; i++)
        msg.push_back((bit_len >> (8 * i)) & 0xFF);

    uint32 A = 0x67452301;
    uint32 B = 0xefcdab89;
    uint32 C = 0x98badcfe;
    uint32 D = 0x10325476;

    uint32 s[64] = {
        7,12,17,22,7,12,17,22,7,12,17,22,7,12,17,22,
        5,9,14,20,5,9,14,20,5,9,14,20,5,9,14,20,
        4,11,16,23,4,11,16,23,4,11,16,23,4,11,16,23,
        6,10,15,21,6,10,15,21,6,10,15,21,6,10,15,21
    };

    uint32 K[64];
    for (int i = 0; i < 64; i++)
        K[i] = floor(abs(sin(i + 1)) * pow(2, 32));

    for (int offset = 0; offset < msg.size(); offset += 64) {
        uint32 *M = (uint32*)(&msg[offset]);

        uint32 a = A, b = B, c = C, d = D;

        for (int i = 0; i < 64; i++) {
            uint32 F, g;

            if (i < 16) { F = (b & c) | (~b & d); g = i; }
            else if (i < 32) { F = (d & b) | (~d & c); g = (5 * i + 1) % 16; }
            else if (i < 48) { F = b ^ c ^ d; g = (3 * i + 5) % 16; }
            else { F = c ^ (b | ~d); g = (7 * i) % 16; }

            uint32 temp = d;
            d = c;
            c = b;
            b = b + leftrotate(a + F + K[i] + M[g], s[i]);
            a = temp;
        }

        A += a;
        B += b;
        C += c;
        D += d;
    }

    char result[33];
    sprintf(result, "%08x%08x%08x%08x", A, B, C, D);
    return string(result);
}

int main() {
    int server_fd, new_socket;
    struct sockaddr_in address;

    server_fd = socket(AF_INET, SOCK_STREAM, 0);

    address.sin_family = AF_INET;
    address.sin_addr.s_addr = INADDR_ANY;
    address.sin_port = htons(8080);

    bind(server_fd, (struct sockaddr*)&address, sizeof(address));
    listen(server_fd, 3);

    cout << "Server Waiting...\n";

    new_socket = accept(server_fd, NULL, NULL);

    while (true) {
        char buffer[2048] = {0};
        read(new_socket, buffer, 2048);

        string data = buffer;

        int pos = data.find("|");
        string msg = data.substr(0, pos);
        string recv_hash = data.substr(pos + 1);

        if (msg == "exit")
            break;

        cout << "\nMessage from Client: " << msg << endl;

        string server_hash = md5(msg);

        cout << "Client Hash: " << recv_hash << endl;
        cout << "Server Computed Hash: " << server_hash << endl;

        if (server_hash == recv_hash)
            cout << "Integrity Verified\n";
        else
            cout << "Integrity Failed\n";

        string reply;
        cout << "\nEnter message for Client: ";
        getline(cin, reply);

        if (reply == "exit") {
            string send_data = "exit|" + md5("exit");
            send(new_socket, send_data.c_str(), send_data.length(), 0);
            break;
        }

        string hash = md5(reply);
        string send_data = reply + "|" + hash;

        send(new_socket, send_data.c_str(), send_data.length(), 0);
    }

    close(new_socket);
    close(server_fd);
}

int main() {
    int sock;
    struct sockaddr_in serv_addr;

    sock = socket(AF_INET, SOCK_STREAM, 0);

    serv_addr.sin_family = AF_INET;
    serv_addr.sin_port = htons(8080);
    inet_pton(AF_INET, "127.0.0.1", &serv_addr.sin_addr);

    connect(sock, (struct sockaddr*)&serv_addr, sizeof(serv_addr));

    while (true) {
        string message;
        cout << "\nEnter message for Server: ";
        getline(cin, message);

        string hash = md5(message);
        string data = message + "|" + hash;

        send(sock, data.c_str(), data.length(), 0);

            break;

            break;

        string client_hash = md5(recv_msg);

    }

    close(sock);
}
