#include <iostream>
#include <string>
#include <unistd.h>
#include <netinet/in.h>
#include <sys/socket.h>
#include <sys/select.h>
#include <openssl/des.h>
using namespace std;

string hexify(const string& s) {
    string h;
    char buf[3];
    for (unsigned char c : s) {
        sprintf(buf, "%02X", c);
        h += buf;
    }
    return h;
}

string unhexify(const string& h) {
    string s;
    for (int i = 0; i < h.size(); i += 2) {
        string x = h.substr(i, 2);
        s += char(strtol(x.c_str(), NULL, 16));
    }
    return s;
}

string desOp(string text, string keyStr, int mode) {
    while (text.size() % 8) text += ' ';
    DES_cblock key;
    memcpy(key, keyStr.c_str(), 8);
    DES_key_schedule ks;
    DES_set_key_unchecked(&key, &ks);

    string out(text.size(), '\0');
    for (int i = 0; i < text.size(); i += 8) {
        DES_cblock in, ob;
        memcpy(in, text.c_str() + i, 8);
        DES_ecb_encrypt(&in, &ob, &ks, mode);
        memcpy(&out[i], ob, 8);
    }
    return out;
}

bool sendLine(int s, string msg) {
    msg += '\n';
    return send(s, msg.c_str(), msg.size(), 0) > 0;
}

bool recvLine(int s, string& msg) {
    msg.clear();
    char ch;
    while (true) {
        int n = recv(s, &ch, 1, 0);
        if (n <= 0) return false;
        if (ch == '\n') break;
        msg += ch;
    }
    return true;
}

int main() {
    int serverSocket = socket(AF_INET, SOCK_STREAM, 0);

    sockaddr_in addr{};
    addr.sin_family = AF_INET;
    addr.sin_port = htons(8080);
    addr.sin_addr.s_addr = INADDR_ANY;

    bind(serverSocket, (sockaddr*)&addr, sizeof(addr));
    listen(serverSocket, 5);

    string key = "mydeskey";

    cout << "=== DES SERVER STARTED ===\n";

    fd_set readfds;
    while (true) {
        FD_ZERO(&readfds);
        FD_SET(serverSocket, &readfds);
        FD_SET(STDIN_FILENO, &readfds);

        select(max(serverSocket, STDIN_FILENO) + 1, &readfds, NULL, NULL, NULL);

        if (FD_ISSET(STDIN_FILENO, &readfds)) {
            string cmd;
            getline(cin, cmd);
            if (cmd == "exit") {
                close(serverSocket);
                return 0;
            }
        }

        if (FD_ISSET(serverSocket, &readfds)) {
            int clientSocket = accept(serverSocket, NULL, NULL);
            cout << "Client connected\n";

            while (true) {
                string hexCipher;
                if (!recvLine(clientSocket, hexCipher)) break;

                string cipher = unhexify(hexCipher);
                string plain = desOp(cipher, key, DES_DECRYPT);

                cout << "\n[CLIENT MESSAGE RECEIVED]\n";
                cout << "Encrypted (Hex) : " << hexCipher << endl;
                cout << "Decrypted Text  : " << plain << endl;

                string reply;
                cout << "Enter reply: ";
                getline(cin, reply);

                string enc = desOp(reply, key, DES_ENCRYPT);
                string hexReply = hexify(enc);

                cout << "\n[SERVER SIDE]\n";
                cout << "Original Text   : " << reply << endl;
                cout << "Encrypted (Hex) : " << hexReply << endl;

                if (!sendLine(clientSocket, hexReply)) break;
            }

            close(clientSocket);
        }
    }
}

    connect(clientSocket, (sockaddr*)&addr, sizeof(addr));

    string key = "mydeskey";

    cout << "=== DES CLIENT STARTED ===\n";

    while (true) {
        string msg;

        string decReply = desOp(encReply, key, DES_DECRYPT);

        cout << "Encrypted (Hex) : " << hexReply << endl;
        cout << "Decrypted Text  : " << decReply << endl;

}
